#include <iostream>
using namespace std;

int final[41] = { 0, };

int fibonacci(int n) {
	if (n == 0) {
		final[0] = 0;
		return 0;
	}
	else if (n == 1) {
		final[1] = 1;
		return 1;
	}
	if (final[n] != 0) //if it is entered, return value
		return final[n];
	else { //if it is not entered, return value
		return final[n] = fibonacci(n - 1) + fibonacci(n - 2);
	}
}

int main() {
	int T;
	int *N;

	cin >> T;
	N = new int[T];
	
	for (int i = 0; i < T; i++)
		cin>>N[i];

	for (int i = 0; i < T; i++) {
		
		if (N[i] == 0)
			cout << 1 << " " << 0 << endl;
		else if (N[i] == 1)
			cout << 0 << " " << 1 << endl;
		else {
			fibonacci(N[i]);
			cout << final[N[i] - 1] << " " << final[N[i]] << endl;
		}
	}
	return 0;
}
